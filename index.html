// === AUDIO (Web Audio API pro lepší podporu iOS) === // (Base64 stringy SILENCE_WAV a CHIME_WAV a TICK_WAV si ponech z původního kódu) const AudioContext = window.AudioContext || window.webkitAudioContext; const audioCtx = new AudioContext(); let chimeBuffer = null; let tickBuffer = null; // Funkce pro dekódování base64 audia do bufferu async function loadSound(base64Data) { try { const response = await fetch(base64Data); const arrayBuffer = await response.arrayBuffer(); return await audioCtx.decodeAudioData(arrayBuffer); } catch (e) { console.error("Chyba při načítání zvuku:", e); } } // Přednačtení zvuků do paměti loadSound(CHIME_WAV).then(buffer => chimeBuffer = buffer); loadSound(TICK_WAV).then(buffer => tickBuffer = buffer); let audioUnlocked = false; function unlockAudio() { if (audioUnlocked) return; // Probuzení AudioContextu na základě interakce uživatele (nutnost pro iOS) if (audioCtx.state === 'suspended') { audioCtx.resume(); } // Přehrání tichého oscilátoru pro definitivní odemčení audio session v Safari const oscillator = audioCtx.createOscillator(); const gainNode = audioCtx.createGain(); gainNode.gain.value = 0; // Zcela potichu oscillator.connect(gainNode); gainNode.connect(audioCtx.destination); oscillator.start(0); oscillator.stop(0.1); audioUnlocked = true; console.log('Web Audio API odemčeno pro iOS'); } // Odemknutí při PRVNÍ interakci uživatele ['touchstart', 'touchend', 'click', 'mousedown'].forEach(evt => { document.addEventListener(evt, unlockAudio, { once: true, passive: true }); }); // Univerzální funkce pro přehrání bufferu function playSound(buffer, volume) { if (!state.soundEnabled || !buffer) return; // Ujištění, že je kontext aktivní if (audioCtx.state === 'suspended') { audioCtx.resume(); } const source = audioCtx.createBufferSource(); source.buffer = buffer; const gainNode = audioCtx.createGain(); gainNode.gain.value = volume; source.connect(gainNode); gainNode.connect(audioCtx.destination); source.start(0); } function playChime() { playSound(chimeBuffer, 1.0); } function playTick() { playSound(tickBuffer, 0.8); }
